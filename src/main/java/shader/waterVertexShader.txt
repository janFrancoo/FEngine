#version 400 core

in vec2 position;

out vec4 clipSpaceCoord;

uniform mat4 transformationMatrix;
uniform mat4 viewMatrix;
uniform mat4 projectionMatrix;

void main() {
	clipSpaceCoord = projectionMatrix * viewMatrix * transformationMatrix * vec4(position.x, 0.0, position.y, 1.0);
	gl_Position = clipSpaceCoord;
}
